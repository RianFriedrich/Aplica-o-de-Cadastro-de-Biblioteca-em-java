Sistema de cadastro de livros
tablea livros
tabela autor 
relacionamento entre elas 
 
  create table autor(
    id int auto_increment primary key,
    nome varchar(50),
    nacionalidade varchar(50),
    datanasc varchar(50)
    );

create table livro(
    id int auto_increment primary key,
    titulo varchar(50), 
    genero varchar(50),
    anoPub varchar(50),
    id_autor int not null,
    numCop varchar(50),
    foreign key(id_autor) references autor(id)
    );
